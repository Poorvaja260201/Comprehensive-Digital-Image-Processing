from scipy.stats import zscore

# Calculate Z-scores for Years_of_Experience
data['Z_Score_Years'] = zscore(data['Years_of_Experience'])

# Filter out outliers
data = data[abs(data['Z_Score_Years']) <= 3]
data = data.drop(columns=['Z_Score_Years'])


# Data Cleaning
def impute_age(group):
    group['Age'] = group['Age'].fillna(group['Age'].mean())
    return group

# a) Handle missing values
# Impute Age by mean of Education level and Gender
data = data.groupby(['Education', 'Gender'], group_keys=False).apply(impute_age)

# Impute Gender with mode
data['Gender'] = data['Gender'].fillna(data['Gender'].mode()[0])

# Impute Income with median
data['Income'] = data['Income'].fillna(data['Income'].median())

# b) Handle outliers
# Income outliers using IQR
q1_income = data['Income'].quantile(0.25)
q3_income = data['Income'].quantile(0.75)
iqr_income = q3_income - q1_income

income_lower_bound = q1_income - 1.5 * iqr_income
income_upper_bound = q3_income + 1.5 * iqr_income

data = data[(data['Income'] >= income_lower_bound) & (data['Income'] <= income_upper_bound)]

# Years_of_Experience outliers using Z-Score
data['Z_Score_Years'] = zscore(data['Years_of_Experience'])
data = data[abs(data['Z_Score_Years']) <= 3]
data = data.drop(columns=['Z_Score_Years'])

print("Data cleaning completed.")

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression, LogisticRegression
from sklearn.metrics import mean_squared_error, accuracy_score
from sklearn.preprocessing import MinMaxScaler
import numpy as np

# Example dataset
data = pd.DataFrame({
    'Age': [25, 35, 45, 55, 65],
    'Income': [30000, 50000, 70000, 90000, 110000],
    'Education': ['High School', 'Bachelor', 'Master', 'PhD', 'Bachelor'],
    'Department': ['HR', 'IT', 'Finance', 'IT', 'HR']
})

# Step 1: Feature Engineering
scaler = MinMaxScaler()
data[['Age', 'Income']] = scaler.fit_transform(data[['Age', 'Income']])

# One-hot encode categorical variables
data = pd.get_dummies(data, columns=['Education', 'Department'], drop_first=True)

# Create High-Income label based on the top 25%
data['High_Income'] = (data['Income'] >= data['Income'].quantile(0.75)).astype(int)

# Step 2: Split dataset
X = data.drop(columns=['Income', 'High_Income'])
y_income = data['Income']
y_high_income = data['High_Income']

X_train, X_test, y_income_train, y_income_test = train_test_split(X, y_income, test_size=0.3, random_state=42)
_, _, y_high_income_train, y_high_income_test = train_test_split(X, y_high_income, test_size=0.3, random_state=42)

# Step 3: Train Linear Regression Model
linear_model = LinearRegression()
linear_model.fit(X_train, y_income_train)

# Predict and evaluate
y_income_pred = linear_model.predict(X_test)
mse = mean_squared_error(y_income_test, y_income_pred)
print(f"Linear Regression MSE: {mse}")

# Step 4: Train Logistic Regression Model
logistic_model = LogisticRegression()
logistic_model.fit(X_train, y_high_income_train)

# Predict and evaluate
y_high_income_pred = logistic_model.predict(X_test)
accuracy = accuracy_score(y_high_income_test, y_high_income_pred)
print(f"Logistic Regression Accuracy: {accuracy}")
